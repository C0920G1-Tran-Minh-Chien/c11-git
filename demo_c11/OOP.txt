<script>
    class Person {
        constructor( name, age, gender  ) {
            this.name = name ;
            this.age = age ;
            this.gender = gender ;
        }
        // get là lấy giá trị ra.
        getName(){
            return this.name;
        }
        // set là gán lại giá trị cho thuộc tính
        setName( name ) {
            this.name = name;
        }
        getAge(){
            return this.age;
        }
        setAge( age ) {
            this.age = age;
        }

        study() {
            alert( this.name + " không muốn học!");
        }
    }

    // Class Student kế thừa những thuộc tính và hành vi của class Person.
    class Student extends Person {

        constructor(name, age, gender, score ) {
            // Đại diện cho constructor của class Cha
            super(name, age, gender);
            this.score = score;
        }

        // CÓ thể định nghĩa lại hành vi của class cha. gọi là Override
        study(){
            alert(this.name + " Đang học!");
        }

        // study( subject ){
        //     alert(this.name + " Đang học môn " + subject);
        // }
        // super đại diện cho class cha.
        say(){
            super.study();
        }

    }

    let person1 = new Person();
    person1.setName("Chiến") ;
    person1.setAge(30)
    console.log(person1.getAge());

    console.log("Student có gì!");
    // student1.setName("Trung");
